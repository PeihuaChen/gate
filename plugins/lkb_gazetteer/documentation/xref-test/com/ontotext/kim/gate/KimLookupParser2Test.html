169     public final void testSymbolicPrefixSuffixMatter() {
170         aliasCache = new AliasCacheStub(
171                 null, new ArrayList(), KIMConstants.CASE_SENSITIV);
172         KimLookupParser parser = new KimLookupParser(aliasCache);
173         // Alias with symbolic prefix and suffix
174         addAliasCacheEntry("-single-blind-");
175         parser.findLookups("single-blind", occuranceHandler);
176         // +1 lookup for "single-blind" because "single" is valid alias prefix
177         assertEquals(3, dictLookupsCount);
178         // but no valid alias is found because symbolic prefix is not matched
179         assertEquals(0, occuranceHandler.annotatedEntities);
180         dictLookupsCount = 0;
181         parser.findLookups("&*^%%-single-blind-LL-234234", occuranceHandler);
182         // Lookup for "single", "single blind", "LL" and "234234" 
183         assertEquals(5, dictLookupsCount);
184         // Find the alias with symbolic prefix&suffix despite
185         // the larger symbolic prefix in the text
186         assertEquals(2, occuranceHandler.annotatedEntities);
187     }
