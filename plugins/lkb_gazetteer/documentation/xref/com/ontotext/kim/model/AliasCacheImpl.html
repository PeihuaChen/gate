40  /**
41   * This class is designed to serve as an Alias dictionary for the
42   * <code>KimGazetteer</code>. It is used to store an image of the textual
43   * representations (Aliases) of the known objects (Entities) described in the
44   * KIM platform knowledge-base. Then it allows to check text fragments for
45   * the presence of known Aliases.<br>
46   * The aliases are not stored as plain text but as couple of hash codes.
47   * The storage is implemented by the couple of classes <code>HashedAlias</code>
48   * and <code>HashRegister</code>. Because of this specific - before storing
49   * of an Alias and also before checking a text fragment - they both must
50   * be pre-processed. The pre-processing is implemented in couple of classes.
51   * The class <code>AliasTextTransformer</code> does preliminary text
52   * normalization and the class <code>ParsingFrame</code> does further
53   * normalization and hash-codes calculation.<br>
54   * <br>
55   * An instance of the <code>AliasCacheImpl</code> class is obtained through
56   * a synchronized factory method.
57   * 
58   * @author danko
59   *
60   */
61  public class AliasCacheImpl implements AliasLookupDictionary {

107 	 * of the alias cache
108 	 * @param dictionaryPath 
109 	 * @param caseSens 
110 	 * @return - the instance of the cache
111 	 * @throws ResourceInstantiationException
112 	 */
113 	public static AliasCacheImpl getInstance(File dictionaryPath, String caseSens) throws ResourceInstantiationException {
114 		synchronized(instanceLock) {
115 			if ( !aliasDictionaries.containsKey(dictionaryPath)) {     
116 				AliasCacheImpl aliasCache = createInstance(dictionaryPath, caseSens);
117 				aliasDictionaries.put(dictionaryPath, aliasCache);
118 			}
119 		}
120 		return aliasDictionaries.get(dictionaryPath);

140 	/**
141 	 * Checks whether an alias can be added to the cache, honoring the
142 	 * caseSensitivity setting.<br>
143 	 * <br>
144 	 * The ignore list check is performed here.<br>
145 	 * 
146 	 * @param alias
147 	 * @return whether the alias can be added
148 	 */
149 	private boolean verifyAlias(String alias) {
150 		if (alias==null || alias.trim().length() == 0)
151 			return false;
152 
153 		alias = (String) ParsingFrame.frameTT.transform(alias);
154 		if (aliasToIgnore.exists(alias.hashCode(), alias)) {
155 			log.info("'" + alias
156 					+ "' ignored, because it was found in the ignore list.");
157 			return false;
158 		}
159 
160 		return true;
161 	}

164 	 * <code>aliasInstRegister</code> register */
165 	private static class InstanceRegisterLoader
166 	implements HashRegister.ContentProcessor {
167 		HashRegister instRegister;
168 		public InstanceRegisterLoader(HashRegister register) {
169 			instRegister = register;
170 		}
171 		public void process(Object[] elements) {
172 			if (elements != null) {
173 				for (int i=0; i<elements.length; i++) {
174 					String shortInst = ((HashedAlias) elements[i]).shortInstURI;
175 					instRegister.add(shortInst.hashCode(), shortInst);
176 				}
177 			}
178 		}
179 	}

181 	/** This is a blank initialization method. It creates the structure of a new
182 	 * Alias cache without filling it with content. This code is separated from 
183 	 * the data filling from the semantic repository - for class extension
184 	 * convenience.<br>
185 	 * @param ignoreAliases - a String list of aliases to be ignored.
186 	 */
187 	protected void initBlankCache(Collection<String> ignoreAliases) {
188 		aliasRegister = new HashRegister();
189 		aliasPrefixes = new TIntHashSet();
190 		aliasInstRegister = new HashRegister();
191 		instNS = new ArrayList<String>();
192 		classCache = new ArrayList<String>();
193 
194 		// Create a TextTransformer instance for Alias text normalization
195 		Transformer tt = new AliasTextTransformer(
196 				caseSensitivity.equals(KIMConstants.CASE_INSENSITIV));
197 		ParsingFrame.frameTT = tt;
198 		aliasToIgnore = new HashRegister();
199 		if (ignoreAliases != null) {
200 			for (String alias : ignoreAliases) {
201 				// Apply same text normalization to the aliases to be ignored 
202 				alias = (String)tt.transform(alias);
203 				aliasToIgnore.add(alias.hashCode(), alias);
204 				if (caseSensitivity.equals(KIMConstants.CASE_SENSITIV_ALLUPPER)) {
205 					alias = alias.toUpperCase();
206 					aliasToIgnore.add(alias.hashCode(), alias);
207 				}
208 			}
209 		}
210 		log.info(
211 				"Aliases in IGNORE list:" + aliasToIgnore.getElementsCount());
212 
213 	}

215 	/** This method implements the default full initialization process.
216 	 * It creates an empty Alias cache and then fills it with data. The data
217 	 * is collected either from the semantic repository or from a serialization
218 	 * source (a file). 
219 	 * @param ignoreAliases - a String list of aliases to be ignored.
220 	 * @param semRep 
221 	 * @param dictionaryPath 
222 	 * @throws RemoteException - on failure to access the semantic repository.
223 	 */
224 	protected void initCache(
225 			Collection<String> ignoreAliases, QueryResultListener.Feed dataFeed, File dictionaryPath) throws RemoteException {
226 		Profiler pro = new Profiler();
227 		pro.enableGCCalling(false);
228 		pro.printToSystemOut(true);
229 		pro.initRun("Loading of Entities Cache");
230 		pro.checkPoint("start loading");
231 
232 		initBlankCache(ignoreAliases);
233 
234 		File fileTCache = new File(dictionaryPath, "kim.trusted.entities.cache").getAbsoluteFile();
235 		File flagTCache = new File(dictionaryPath, fileTCache.getName() + ".flag");
236 
237 		try {
238 			ensureCachePath(dictionaryPath);
239 		} catch (IOException e1) {
240 		    log.error(
241 					"Could not create entity cache.", e1);
242 		}
243 
244 		boolean flagTLoaded = false;
245 		if (!flagTCache.exists() && fileTCache.exists()) {
246 		    log.info("Loading of trusted entities from "
247 					+ fileTCache);
248 
249 			try {
250 				ObjectInputStream ois = new ObjectInputStream(
251 						new FileInputStream(fileTCache));
252 				Object[] res = (Object[]) ois.readObject();
253 				ois.close();
254 				aliasRegister = (HashRegister) res[0];
255 				aliasPrefixes = (TIntHashSet) res[1];
256 				instNS = (ArrayList<String>) res[2];
257 				classCache = (ArrayList<String>) res[3];
258 
259 
260 				aliasInstRegister = new HashRegister();
261 				// The exactly same Entity InstURI strings are reused
262 				aliasRegister.processContent(
263 						new InstanceRegisterLoader(aliasInstRegister));				
264 				flagTLoaded = true;
265 				log.info(aliasRegister.getElementsCount() + " elements loaded.");
266 			} catch (Exception e) {
267 			    log.error("Loading from "
268 						+ fileTCache + " failed. "
269 						+ "Continue with loading from Semantic Repository.", e);
270 			}
271 		}
272 
273 		if (!flagTLoaded) {
274 			loadTrustedMaps(dataFeed);
275 
276 			try {
277 				flagTCache.createNewFile();
278 				if (fileTCache.exists())
279 					fileTCache.delete();
280 
281 				ObjectOutputStream oos = new ObjectOutputStream(
282 						new FileOutputStream(fileTCache));
283 				oos.writeObject(
284 						new Object[]{aliasRegister, aliasPrefixes,
285 								instNS, classCache});
286 				oos.close();
287 				flagTCache.delete();
288 			} catch (Exception ex) {
289 			    log.error("Saving of trusted entities to "
290 						+ fileTCache + " failed.", ex);
291 			}
292 		}
293 
294 		log.info("Aliases were loaded");
295 		pro.checkPoint("cache loaded");
296 	}

305 	private void loadTrustedMaps(QueryResultListener.Feed dataFeed) {
306 	    log.info("Loading of trusted entities from Sesame");
307 
308 		String filePath = System.getProperty("kim.home.dir", ".") +
309 		EntityPriority.PRIORITY_CONF_FILE.substring(1);
310 		existsClassPriority = (new File(filePath)).exists();
311 
312 		if (existsClassPriority) {
313 			try {
314 				entPrior = new EntityPriority();
315 				entPrior.init();
316 				existsClassPriority = existsClassPriority
317 				&& entPrior.getFilterLookups();
318 			} catch (Exception e) {
319 				log.error(
320 						"Cannot create instance of Priorities class", e);
321 				entPrior = null;
322 			}
323 		}
324 		EntitiesQueryListener entityListener = new TrustedEntitiesListener(entPrior);
325 
326 		// Handler to preserve the same inner listener for the two queries		
327 		if ( log.isDebugEnabled() ) {
328 			entityListener = StatisticListener.wrap(entityListener, "Thrusted Entities");
329 		}
330 		try {
331 			// semRep.evaluateSelectSeRQL(, entityListener);
332 			
333 			dataFeed.feedTo(entityListener);
334 		} catch (KIMQueryException e) {                        
335 			throw new KIMRuntimeException("The loading failed.", e);
336 		} finally {
337 			log.info("The loading from Sesame finished");
338 		}
339 	}

342 	 * used to process the input from the semantic repository. It is used
343 	 * only when the data is loaded from there. */
344 	class TrustedEntitiesListener extends EntitiesQueryListener {
345 		private final EntityPriority m_entPrior;
346 		TrustedEntitiesListener(EntityPriority m_entPrior) {
347 			this.m_entPrior = m_entPrior;
348 		}
349 		@Override
350 		protected void addEntity(String instUri, String classUri,
351 				String aliasLabel) {
352 			addAlias(instUri, classUri, aliasLabel, true);
353 		}
354 
355 		@Override
356 		public void endTableQueryResult() throws IOException {
357 			super.endTableQueryResult();
358 			if (existsClassPriority && allPrioritiesCompetition != null) {
359 				Iterator it = allPrioritiesCompetition.keySet().iterator();
360 				while (it.hasNext()) {
361 					ArrayList<priorityCompetition> pcList =
362 						allPrioritiesCompetition.get(it.next());
363 					int maxPrior = pcList.get(0).maxPriority;
364 					int treshold = m_entPrior.getThreshold();
365 					for (int i = 0; i < pcList.size(); i++) {
366 						priorityCompetition pc = pcList.get(i);
367 						if (i == 0 || maxPrior - pc.maxPriority <= treshold)
368 							addAlias(pc.instURI, pc.classURI, pc.alias, false);
369 					}
370 				}
371 				allPrioritiesCompetition = null;
372 			}
373 		}
374 	}

428 	/** Adds an Alias with its instance and semantic class to the Alias cache.
429 	 * A single call to this method could result in adding several records to
430 	 * the alias cache. This is as a result to the standard Alias enrichment
431 	 * logic which is applied over the given as input alias string.
432 	 * @param instURI - the URI of the Entity instance corresponding to the
433 	 * Alias 
434 	 * @param classURI - the URI of the semantic class
435 	 * @param alias - the string of the alias
436 	 * @param primaryAccess - processing specific flag; if <b>true</b> - forces
437 	 * class priority checks.
438 	 */
439 	public void addAlias(String instURI, String classURI, String alias,
440 			boolean primaryAccess) {
441 
442 		if (checkClassPriority(instURI, classURI, alias, primaryAccess))
443 			return;
444 
445 		String[] enriched = aliasEnrichment(alias);
446 		for (int i=0; i< enriched.length; i++) {
447 			if (verifyAlias(enriched[i]))
448 				simpleAddAlias(instURI, classURI, enriched[i]);
449 		}
450 	}

452 	/** This method performs the standard alias enrichment. It covers
453 	 * cases as variants with and without trailing punctuation. 
454 	 * @param alias - the original alias string.
455 	 * @return array of distinct strings which are accepted as equally valid
456 	 * representation of the related to the Alias - Entity.
457 	 */
458 	private String[] aliasEnrichment(String alias) {
459 		HashSet<String> aliases = new HashSet<String>();
460 		aliases.add(alias);
461 		String[] tmp;
462 
463 		// Enrich with UPPER case versions if needed
464 		if (caseSensitivity.equals(KIMConstants.CASE_SENSITIV_ALLUPPER)) {
465 			tmp = aliases.toArray(new String[0]);
466 			for (int i=0; i<tmp.length; i++) {
467 				String tmpNew = tmp[i].toUpperCase();
468 				if (!tmpNew.equals(tmp[i]))
469 					aliases.add(tmpNew);
470 			}
471 		}
472 
473 		// Enrich with versions with stripped punctual suffix
474 		tmp = aliases.toArray(new String[0]);
475 		for (int i=0; i<tmp.length; i++) {
476 			String tmpNew = StringTransformations.stripPunctAtEnd(tmp[i]);
477 			if (!tmpNew.equals(tmp[i]))
478 				aliases.add(tmpNew);
479 		}
480 
481 		return aliases.toArray(new String[0]);
482 	}

484 	/** Method that implements a simple alias addition (as is) to the
485 	 * cache structures
486 	 * @param instURI - the Entity instance URI
487 	 * @param classURI - the semantic class URI
488 	 * @param alias - the string of the alias
489 	 */
490 	private void simpleAddAlias(String instURI, String classURI,
491 			String alias) {
492 		String shortInstURI = packNS(instURI);
493 
494 		// Calculate different hash values related with the alias
495 		ParsingFrame pfm = new ParsingFrame(alias);
496 		do {
497 			if (pfm.parseOne() && pfm.frameCanExpand()) {
498 				// Add the aliases prefixes to the alias prefix register
499 				Stats.markIt(-1);
500 				aliasPrefixes.add(pfm.getAliasHash1());
501 				Stats.markIt(0);
502 			}
503 		} while (pfm.frameCanExpand());
504 
505 		// Add the corresponding instURI in the Entity instance URI register
506 		String oldURI = null;
507 
508 		// Re-usage of the same string for EntityURI for all its Aliases
509 		// is expected to save 10% to 15% of the memory
510 		oldURI = (String) aliasInstRegister.get(
511 				shortInstURI.hashCode(), shortInstURI);
512 
513 		Stats.markIt(1);
514 
515 		if (oldURI != null)
516 			shortInstURI = oldURI;
517 		else
518 			aliasInstRegister.add(shortInstURI.hashCode(), shortInstURI);
519 		Stats.markIt(2);
520 
521 		// Add the alias in the alias register
522 		aliasRegister.add(
523 				pfm.getAliasHash1(),
524 				new HashedAlias(pfm.getAliasHash2(),
525 						pfm.getPrefixLen(), pfm.getSuffixLen(),
526 						shortInstURI,
527 						packClass(classURI)));
528 		Stats.markIt(3);
529 	}

551 	private ArrayList<KimLookupParser.AliasWrapper> lookup(
552 			ParsingFrame pfm, boolean exactlySame) {
553 		Stats.markIt(-1);
554 		ArrayList<KimLookupParser.AliasWrapper> res = new ArrayList<KimLookupParser.AliasWrapper>();
555 
556 		Object[] tmp = aliasRegister.get(pfm.getAliasHash1());
557 		Stats.markIt(5);
558 		if (tmp == null || tmp.length==0)
559 			return res;
560 
561 		for (int i=0; i<tmp.length; i++) {
562 			HashedAlias ha = (HashedAlias)tmp[i];
563 			if (!exactlySame) {
564 				pfm.setNewPrefSufLen(ha.prefLen, ha.suffLen);
565 			}
566 			if (pfm.getAliasHash2() == ha.aliasHash2) {
567 				res.add(new KimLookupParser.AliasWrapper(
568 						unpackNS(ha.shortInstURI),
569 						unpackClass(ha.classID),
570 						pfm.getAliasStart(), pfm.getAliasEnd()));
571 			}
572 		}
573 		Stats.markIt(6);
574 
575 		return res;
576 	}

596 	private String packNS(String fullURI) {
597 		String ns, ln;
598 		try {
599 			URI uri = new URIImpl(fullURI);
600 			ns = uri.getNamespace();
601 			ln = uri.getLocalName();
602 		} catch (RuntimeException e) {
603 			ns = "";
604 			ln = fullURI;
605 		}
606 		int j = instNS.indexOf(ns);
607 		if (j < 0) {
608 			instNS.add(ns);
609 			j = instNS.size()-1;
610 		}
611 		return j+":"+ln;
612 	}
613 	private String unpackNS(String shortURI) {
614 		if (shortURI == null)  return null;
615 		int i = shortURI.indexOf(':');
616 		String ns;
617 		try {
618 			ns = instNS.get(Integer.parseInt(shortURI.substring(0,i)));
619 		} catch (Exception e) {
620 			log.debug(
621 					"Short URI unpack failed:" + shortURI, e);
622 			return shortURI;
623 		}
624 		return ns + shortURI.substring(i+1);
625 	}
626 	private short packClass(String classURI) {
627 		int i = classCache.indexOf(classURI);
