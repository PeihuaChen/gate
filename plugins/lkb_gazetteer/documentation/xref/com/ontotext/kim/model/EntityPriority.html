41      public void init() throws Exception {
42          m_prop.load(new FileInputStream(new File(System.getProperty(
43                  "kim.home.dir", "."), PRIORITY_CONF_FILE.substring(1))));
44          m_nDefaultInstancePriority = Integer.parseInt(m_prop
45                  .getProperty("priority.default.instance"));
46          m_nDefaultClassPriority = Integer.parseInt(m_prop
47                  .getProperty("priority.default.class"));
48          m_nThreshold = Integer.parseInt(m_prop
49                  .getProperty("priority.threshold"));
50          m_bFilterLookups = (m_prop.getProperty("priority.filterLookups") == null) ? false
51                  : Boolean.parseBoolean(m_prop
52                          .getProperty("priority.filterLookups"));
53  
54          // init class and instance priority
55          for (Iterator iter = m_prop.keySet().iterator(); iter.hasNext();) {
56              String item = (String) iter.next();
57              if (item.startsWith(PRIORITY_CLASS_PREFIX))
58                  addClassPriority(item);
59              else if (item.startsWith(PRIORITY_INSTANCE_PREFIX))
60                  addInstancePriority(item);
61          }
62  
63          // init pair-class priority
64          for (int i = 1; true; i++) { //
65              String sRuleName = m_prop.getProperty(PRIORITY_RULE_PREFIX + i
66                      + ".name");
67              if (sRuleName == null)
68                  break;
69              String sClass1 = m_prop.getProperty(PRIORITY_RULE_PREFIX + i
70                      + ".class1");
71              String sClass2 = m_prop.getProperty(PRIORITY_RULE_PREFIX + i
72                      + ".class2");
73              int nDelta = Integer.parseInt(m_prop
74                      .getProperty(PRIORITY_RULE_PREFIX + i + ".delta"));
75              if (sClass1 == null || sClass2 == null)
76                  throw new RuntimeException(
77                          "Invalid priority rule definition, rule " + i);
78              m_hRules.put(sClass1 + "_" + sClass2, new Integer(nDelta));
79              m_hRules.put(sClass2 + "_" + sClass1, new Integer(-nDelta));
80          }
81      }
82  
83      protected void addClassPriority(String sItemToParse) {
84          String sClass = sItemToParse.substring(PRIORITY_CLASS_PREFIX.length());
85          Integer nPrio = new Integer(m_prop.getProperty(sItemToParse));
86          m_hClassPrio.put(sClass, nPrio);
87      }

89      protected void addInstancePriority(String sItemToParse) {
90          String sInst = sItemToParse

93          m_hInstPrio.put(sInst, nPrio);
94      }

106     public int getPairDelta(String sClass1, String sClass2) {
107         Integer nDelta = (Integer) m_hRules.get(stripNameSpace(sClass1) + "_"

109         if (nDelta == null)
110             return 0;

122     /**
123      * Strips the namespace from the URI and retains only the local name.
124      * 
125      * @param sLongURI
126      *            an URI
127      * @return the localname of the URI
128      * @throws KIMException

130     public static String stripNameSpace(String sLongURI) {
131         String result;
132         int index = sLongURI.indexOf('#');
133         if (index < 0) {
134             return sLongURI;
135         }
136         result = sLongURI.substring(index + 1);
137         return result;
