12  /**
13   * This class implements a hash-code indexed storage for objects. It resembles
14   * the <code>HashSet</code> class with these substantial differences:<br>
15   * - The hash-code value is supplied externally
16   * - Multiple objects can be stored with equal hash-codes. They are grouped
17   * and stored in packages. These packages are implemented as the internal
18   * class <code>HashElement</code>.<br>
19   * - Stored objects are retrieved by packages through their hash-code value.<br>
20   * - If stored objects implement <code>Comparable</code> they are sorted in the
21   * packages and are returned as a sorted array. This allows binary search within
22   * the result.<br>
23   * <br>
24   * The implementation of the storage structure as also the resize logic is
25   * simplified intentionally. By design it is intended to support only operations
26   * addition, search and retrieval (not removal).
27   * 
28   * @author danko
29   */
30  public class HashRegister implements Serializable {
31      private static final long serialVersionUID = 2442L;
32  
33      /**
34       * This flag could take the following values:<br>
35       * 0 - add the duplicating object;<br>
36       * 1 - replace the old object and put the new one;<br>
37       * 2 - keep the old one and ignore the new one;<br>
38       */
39      protected static int todoIfExists = 1;
40      /**
41       * Determines the step for extension of the array field
42       * <code>HashElement.elementHolder</code>
43       */
44      protected static int subRegIncrement = 2;

48  	 * Determines the initial size of the main-register. A better distribution
49  	 * is achieved if it is a prime number.

53  	 * Determines the maximal percent of filling of the main-register
54  	 * after which the HashRegister is expanded by doubling its size.
55  	 */
56  	protected int maxFillPerc = 75;

69  	public static class HashElement implements Serializable {
70  	    private static final long serialVersionUID = 2442L;

72  	    public final int subRegHash;

79  		/**
80  		 * Adds new element to the container. The added element must have
81  		 * the same hash-code value as the container. This must be verified
82  		 * outside this class.
83  		 * 
84  		 * @param element - the added element
85  		 */
86          public byte add(Serializable element) {
87              byte elemCountChange = add12(element); 
88              if (elemCountChange < 0) {
89                  if (element instanceof Comparable)
90                      elemCountChange = addSorted((Comparable) element);
91                  else
92                      elemCountChange = addUnsorted(element);
93              }
94              
95              return elemCountChange;
96          }
97          /** Method adds new element to the container in unsorted manner.
98           * The method is called only if the container already have 2 
99           * elements stored in. 
100          * @param element - the element to store
101          * @return number of elements stored: 1 - if stored successfully;
102          * 0 - element already existed and no duplicates are allowed;
103          */
104         private byte addUnsorted(Object element) {
105             if (todoIfExists > 0) {
106                 int i = Arrays.asList(elementHolder).indexOf(element);
107                 if (i >= 0)
108                     switch (todoIfExists) {
109                     case 1:  // replace the old object and put the new one
110                         ((Object[])elementHolder)[i] = element;
111                         return 0;
112                     case 2:  // drop the new one and retain the old one
113                         return 0;
114                     }
115             }
116             
117             Object[] newSubReg;
118             Object[] oldSubReg = ((Object[])elementHolder);
119             int addPosition = oldSubReg.length;
120             while (oldSubReg[addPosition - 1] == null)
121                 addPosition--;
122             if (addPosition == oldSubReg.length) {
123                 newSubReg = new Object[oldSubReg.length + subRegIncrement];
124                 for (int i=0; i < oldSubReg.length; i++)
125                     newSubReg[i] = oldSubReg[i];
126             }
127             else
128                 newSubReg = oldSubReg;
129             newSubReg[addPosition] = element;
130             elementHolder = newSubReg;
131             return 1;
132         }
133         
134         private static final Comparator<Comparable> nsc = new NullComparator(true);
135         
136         /** Method adds new element to the container in sorted manner.
137          * The method is called only if the container already have 2 
138          * elements stored in. 
139          * @param element - the element to store
140          * @return number of elements stored: 1 - if stored successfully;
141          * 0 - element already existed and no duplicates are allowed;
142          */
143         private byte addSorted(Comparable element) {
144             Comparable[] newSubReg;
145             Comparable[] oldSubReg = ((Comparable[])elementHolder);
146             int addPosition = Arrays.binarySearch( oldSubReg, element, nsc);
147             if (todoIfExists > 0 && addPosition >= 0) {
148                 switch (todoIfExists) {
149                 case 1:  // replace the old object and put the new one
150                     ((Object[])elementHolder)[addPosition] = element;
151                     return 0;
152                 case 2:  // drop the new one and retain the old one
153                     return 0;
154                 }
155             }
156             if (addPosition < 0)
157                 addPosition = - addPosition - 1;
158             if (oldSubReg[oldSubReg.length-1] != null) {
159                 newSubReg = new Comparable[oldSubReg.length + subRegIncrement];
160                 for (int i = 0; i < oldSubReg.length; i++) {
161                     if (i < addPosition)
162                         newSubReg[i] = oldSubReg[i];
163                     else
164                         newSubReg[i+1] = oldSubReg[i];
165                 }
166             }
167             else {
168                 for (int i = oldSubReg.length-1; i > addPosition; i--)
169                     oldSubReg[i] = oldSubReg[i-1];
170                 newSubReg = oldSubReg;
171             }
172             
173             newSubReg[addPosition] = element;
174             elementHolder = newSubReg;
175             return 1;
176         }
177         
178         /** Method adds new element to the container in sorted manner.
179          * The method is called only if the container is empty or has 1 
180          * element stored in.
181          * @param element - element to be stored
182          * @return - returns the change in the number of stored elements
183          * (0 or 1) or returns -1 if element is not handled
184          */
185         private byte add12(Serializable element) {
186             if (element == null)
187                 return 0;
188             
189             if (elementHolder == null) {
190                 elementHolder = element;
191                 return 1;
192             }
193             else if ( !(elementHolder instanceof Object[])){
194                 if (todoIfExists > 0 && element.equals(elementHolder)) {
195                     switch (todoIfExists) {
196                     case 1:  // replace the old object and put the new one
197                         elementHolder = element;
198                         return 0;
199                     case 2:  // drop the new one and retain the old one
200                         return 0;
201                     }
202                 }
203                 Object[] newSubReg;
204                 if (element instanceof Comparable) {
205                     newSubReg = new Comparable[1 + subRegIncrement];
206                     if (nsc.compare((Comparable) elementHolder,
207                             (Comparable) element) < 0) {
208                         newSubReg[0] = elementHolder;
209                         newSubReg[1] = element;
210                     }
211                     else {
212                         newSubReg[0] = element;
213                         newSubReg[1] = elementHolder;
214                     }
215                 }
216                 else {
217                     newSubReg = new Object[1 + subRegIncrement];
218                     newSubReg[0] = elementHolder;
219                     newSubReg[1] = element;
220                 }
221                 elementHolder = newSubReg;
222                 return 1;
223             }
224             return -1;
225         }

237         /** Checks if given element is already stored in the container.
238          * @param element - the element to be checked.
239          * @return <b>true</b> - if found
240          */
241         public boolean contains(Object element) {
242             Object[] elements = getElements();
243             if (element == null)
244                 return (elements == null);
245             else if (element instanceof Comparable)
246                 return Arrays.binarySearch( (Comparable[])elements,
247                         (Comparable) element, nsc) >= 0;
248             else
249                 return Arrays.asList(elements).contains(element);
250         }
251         
252         /** Gets the stored instance that equals the given instance.
253          * @param element - the element to be checked.
254          * @return the equal stored instance
255          */
256         public Object get(Object element) {
257             Object[] elements = getElements();
258             int i = -1;
259             if (element != null) {
260                 if (element instanceof Comparable)
261                     i = Arrays.binarySearch( (Comparable[])elements,
262                             (Comparable) element, nsc);
263                 else
264                     i = Arrays.asList(elements).indexOf(element);
265             }
266             return ((i >= 0)? elements[i]: null);
267         }

269 		/** Returns the stored elements as an object array
270 		 * @return array with all stored elements
271 		 */
272 		public Object[] getElements() {
273             if (elementHolder == null)
274                 return null;
275             else if (elementHolder instanceof Object[]) {
276                 Object[] eList = (Object[]) elementHolder;
277                 int i = eList.length;
278                 while (i>0 && eList[i-1] == null)
279                     i--;
280                 if (eList[0] instanceof Comparable)
281                     return Arrays.asList(
282                             eList).subList(0, i).toArray(new Comparable[0]);
283                 else
284                     return Arrays.asList(
285                             eList).subList(0, i).toArray();
286             }
287             else {  // getPoint.elementHolder contains single object
288                 if (elementHolder instanceof Comparable)
289                     return new Comparable[]{(Comparable) elementHolder};
290                 else
291                     return new Object[]{elementHolder};
292             }
293         }

297 	/** Comparator for sorting <code>HashElement</code> instances. */
298 	public static class HashElementComp implements Comparator<HashElement> {
299 		Object[] container;
300 		public HashElementComp(Object[] container) {
301 			this.container = container;
302 		}
303 		
304 		public int compare(HashElement o1, HashElement o2) {
305 			int k1 = o1.index(container);
306 			int k2 = o2.index(container);
307 			if (k1 < k2)  return -1;
308 			if (k1 > k2)  return 1;
309 			return 0;
310 		}
311 	}

314 	 * which an object will be stored is calculated through the method
315 	 * <code>mainIx</code>. Calculation is based on the externally
316 	 * provided hash-code of the object and the current size of the array */
317 	protected Object[] mainReg;
318 	/** Access synchronization locking object */
319 	protected transient Object mainRegLock = new Object();
320     /** Number of different hash-code values provided with stored objects */
321     protected int hashesCount = 0;
322     /** Current limit of the number of distinct hash-code values that 
323      * could be stored with the current size of <code>mainReg</code> */
324     protected int hashesCountLimit = 0;
325     /** The number of objects stored in the register */
326 	protected int elementsCount = 0;
327 	/** This flag determines if the auto-resizing is turned on */
328 	protected boolean autoResize = true;

339 	 */
340 	protected HashRegister(int newSize, boolean autoRes) {
341 		mainReg = new Object[newSize];
342         hashesCountLimit = mainReg.length * maxFillPerc / 100;
343 		autoResize = autoRes;

346 	/** The method resizes the <code>mainReg</code> array if the content
347 	 * volume limit was reached.
348 	 * @return <b>true</b> if resize was performed
349 	 */
350 	public boolean resizeMainReg() {
351 		// If limit is not reached -> quit resize
352 		if (hashesCountLimit > hashesCount)  return false;
353 		
354 		long start = System.currentTimeMillis();
355 
356 		Object[] newMainReg;
357 		try {
358 			newMainReg = new Object[mainReg.length*2];
359 		} catch (Exception e) {
360 			KimLogs.logNERC_GAZETTEER.warn("Unsuccessful attempt to resize" +
361 					" [HashRegister] due to:\n" + e.getMessage());
362 			return false;
363 		}
364 
365 		synchronized (mainRegLock) {
366 			// Cycle through the old register and move the [HashElement] objects
367 			// to the new main-register
368 			for (int main_i=0; main_i < mainReg.length; main_i++) {
369 				int new_main_i = main_i + mainReg.length;
370 
371 				if (mainReg[main_i] == null) {
372 					// If there was no sub-register:
373 					// Do nothing
374 				}
375 				else if (mainReg[main_i] instanceof HashElement) {
376 					// If there was just one [HashElement] for sub-register:
377 					// Determine it's new place
378 					newMainReg[((HashElement)mainReg[main_i]).index(newMainReg)] =
379 						mainReg[main_i];
380 				}
381 				else {
382 					// If there were two or more [HashElement] as sub-register:
383 					// Try to split the sub-register in two parts.
384 					HashElement[] oldSubReg = (HashElement[]) mainReg[main_i];
385 					
386 					// The comparator [HashElementComp] allows [HashElement]
387 					// instances to be sorted with respect to the value of their
388 					// next main-register index (after the expansion).
389 					Arrays.sort(oldSubReg, new HashElementComp(newMainReg));
390 					
391 					// Scan where is the splitting point between the elements
392 					// that retain their old main-register index and those
393 					// that will move to a new main-register index
394 					int j1 = 0;
395 					while (j1 < oldSubReg.length
396 							&& oldSubReg[j1].index(newMainReg) == main_i)
397 						j1++;
398 					
399 					// Handle the elements retaining the old main-register index
400 					if (j1 == 0) {
401 					}
402 					else if (j1 == 1) {
403 						newMainReg[main_i] = oldSubReg[0];
404 					}
405 					else if (j1 < oldSubReg.length) {
406 						HashElement[] newSubReg = new HashElement[j1];
407 						for (int k=0; k<j1; k++)
408 							newSubReg[k] = oldSubReg[k];
409 						newMainReg[main_i] = newSubReg;
410 					}
411 					else {
412 						newMainReg[main_i] = oldSubReg;
413 					}
414 
415 					// Handle the elements moving to the new main-register index
416 					int j2 = oldSubReg.length - j1;
417 					if (j2 == 0) {
418 					}
419 					else if (j2 == 1) {
420 						newMainReg[new_main_i] = oldSubReg[oldSubReg.length - 1];
421 					}
422 					else if (j2 < oldSubReg.length) {
423 						HashElement[] newSubReg = new HashElement[j2];
424 						for (int k=0; k<j2; k++)
425 							newSubReg[k] = oldSubReg[j1 + k];
426 						newMainReg[new_main_i] = newSubReg;
427 					}
428 					else {
429 						newMainReg[new_main_i] = oldSubReg;
430 					}
431 				}
432 				
433 				// Release the old sub-register to allow for early GC.
434 				mainReg[main_i] = null;
435 			}
436 			
437 			// Release the old main-register to allow for GC.
438 			mainReg = newMainReg;
439 	        hashesCountLimit = mainReg.length * maxFillPerc / 100;
440 		}
441 		long duration = System.currentTimeMillis() - start;
442         KimLogs.logNERC_GAZETTEER.debug("Main Register resized to " +
443                 mainReg.length + " for " + duration + "ms.");
444 		return true;
445 	}

512 	/** This method searches a <code>HashElement</code> corresponding to the
513 	 * given hash-code value. Depending on the flag <code>createIfMissing</code>
514 	 * if no such hash element is found it can be created.
515 	 * @param hashValue - the hash-code value
516 	 * @param createIfMissing - flag to force creation if not found.
517 	 * @return the found/created <code>HashElement</code>
518 	 */
519 	private HashElement getHashElement(
520 			int hashValue, boolean createIfMissing) {
521 		HashElement getPoint;
522 
523 		synchronized (mainRegLock) {
524 
525 			int main_i = mainIx(hashValue, mainReg);
526 			Object subReg = mainReg[main_i];
527 	
528 			if (subReg == null) {
529 				if (createIfMissing) {
530 					getPoint = new HashElement(hashValue);
531 					mainReg[main_i] = getPoint;
532 					hashesCount++;
533 				}
534 				else
535 					getPoint = null;
536 			}
537 			else if (subReg instanceof HashElement) {
538 				getPoint = (HashElement) subReg;
539 				if (getPoint.subRegHash != hashValue)
540 					if (createIfMissing) {
541 						HashElement[] newSubReg = new HashElement[2];
542 						newSubReg[0] = getPoint;
543 						getPoint = new HashElement(hashValue);
544 						newSubReg[1] = getPoint; 
545 						mainReg[main_i] = newSubReg;
546 						hashesCount++;
547 					}
548 					else {
549 						getPoint = null;
550 					}
551 			}
552 			else { // subReg is instance of HashElement[]
553 				HashElement[] oldSubReg = (HashElement[]) subReg;
554 				getPoint = null;
555 				for (int i=0; i < oldSubReg.length; i++) {
556 					if (oldSubReg[i].subRegHash == hashValue) {
557 						getPoint = oldSubReg[i];
558 						break;
559 					}
560 				}
561 				if (createIfMissing && getPoint == null) {
562 					HashElement[] newSubReg = new HashElement[oldSubReg.length+1];
563 					for (int i=0; i<oldSubReg.length; i++)
564 						newSubReg[i] = oldSubReg[i];
565 					getPoint = new HashElement(hashValue);
566 					newSubReg[newSubReg.length-1] = getPoint;
567 					mainReg[main_i] = newSubReg;
568 					hashesCount++;
569 				}
570 			}
571 		}
572 		
573 		return getPoint;
574 	}

581 	/** This is a listener interface which provides means to perform bulk
582 	 * operations over all of the stored elements in the register. This
583 	 * interface must be used together with the method
584 	 * <code>processContent</code>. */

587 	}
588 	
589 	/** This method accepts an implementation of the
590 	 * <code>ContentProcessor</code> interface. Then it cycles all
591 	 * of the stored elements in the register and passes them to the
592 	 * <code>process</code> method of the <code>ContentProcessor</code>
593 	 * implementation.
594 	 * @param cProc - <code>ContentProcessor</code> implementation;
595 	 */
596 	public void processContent(ContentProcessor cProc) {
597 	    for (int main_i=0; main_i<mainReg.length; main_i++) {
598 	        if (mainReg[main_i] == null)  continue;
599 	        if (mainReg[main_i] instanceof HashElement)
600 	            cProc.process(((HashElement)mainReg[main_i]).getElements());
601 	        else {
602 	            HashElement[] subReg = (HashElement[]) mainReg[main_i];
603                 for (int i=0; i<subReg.length; i++)
604                     cProc.process(subReg[i].getElements());
605 	        }
606 	    }
607 	}

613 		int repLen = Math.min(mainReg.length, mainRegLim);
614 		for (int i=0; i<mainReg.length; i++) {
615 			if ( mainReg[i] == null)
616 				;
617 			else if (mainReg[i] instanceof HashElement) {
618 				System.out.println("<"+i+">"+getHEContent((HashElement) mainReg[i]));
619 				repLen--;
620 			}
621 			else {
622 				int subRepLen = Math.min(((HashElement[])mainReg[i]).length, subRegLim);
623 				System.out.print("<"+i+">");
624 				for (int j=0; j<subRepLen; j++)
625 					System.out.print(getHEContent(((HashElement[])mainReg[i])[j])+",");
626 				System.out.println();
627 				repLen--;
628 			}
629 			if (repLen <= 0)  break;
630 		}

641 	public void printDistribution() {
642 		int[] bucketSize = new int[11];
643
