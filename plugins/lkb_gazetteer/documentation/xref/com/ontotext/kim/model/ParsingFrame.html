13  /**
14   * This class is used as a text parsing tool by classes
15   * <code>AliasCacheImpl</code> and <code>KimLookupParser</code> and produces
16   * special output for the class <code>HashedAlias</code>.<br>
17   * The class provides means to parse a string into sequence of plain numeric
18   * and letter lexemes delimited by optional non-alpha-numeric character
19   * sequences of arbitrary length. This is the base for creation of a Hashed
20   * Alias and also the base for searching it in a text.<br>
21   * The class offers a frame view over the parsed string. The frame spans one or
22   * more sequential alpha-numeric lexemes (ANL) and their non-alpha-numeric
23   * context (surrounding puctuation and white spaces).<br>
24   * The frame contains three parts:<br>
25   *  * middle - this is a part of the initial string which starts in the 
26   * beginning of the first ANL of the frame and ends in the end of the last ANL
27   * of the frame.<br>
28   *  * prefix - this is the non-alpha-numeric sequence before the first ANL of
29   * the frame.<br>
30   *  * suffix - this is the non-alpha-numeric sequence after the last ANL of the
31   * frame.<br>
32   * The class has couple of methods for manipulating the frame. The first one
33   * is for extending the frame by one lexeme. The second one is for moving
34   * the frame start by one lexeme, which resets the frame length to one
35   * lexeme.<br>
36   * The class has retrieval methods that get the current frame offset and
37   * the lengths of the three parts. There are dedicated methods for calculation
38   * of the two Hash Codes specific to a Hashed Alias.
39   * 
40   * @author danko
41   *
42   */

49  	/** This is the Reg-Ex pattern that is used search the ANLs */
50  	private static final Pattern LEXEME_MATCH_PAT = Pattern.compile(
51  			"(" +
52  			// Number only lexemes
53  			"(?:\\d)+|" +
54  			
55  			(SPLIT_AT_SMALL_TO_CAPITAL_CASE_CHANGE 
56  					? "(?:\\p{javaUpperCase}{0,}\\p{javaLowerCase}{1,})|(?:\\p{javaUpperCase}{1,})"			    
57  					: "(?:[\\p{javaUpperCase}\\p{javaLowerCase}]+)" ) + 
58  			")"
59  			// Non-Alpha-Numeric suffix
60  			+ "([^\\d\\p{javaUpperCase}\\p{javaLowerCase}]*)"
61  	);

63  	 *  of the parsed string. The normalized for is used to calculate
64  	 *  the H1 hash value of the <code>HashAlias</code> */
65  	private static final String H1_DELIMITER = "^";

74  	private class ParsingBufferElement {

77  		public final int suff;

111 	// ParsingFrame: class constructor

113 	/** The constructor parses the source string with the regular expression
114 	 * based <code>java.util.regex.Matcher</code> and stores the results in
115 	 * the parsing buffer <code>parsingBuffer</code>.
116 	 * @param source - the String to be processed through the
117 	 * <code>ParsingFrame</code> class
118 	 */
119 	public ParsingFrame (String source) {
120 		Stats.markIt(-1);
121 		this.source = source;
122 		Matcher lexMatcher = LEXEME_MATCH_PAT.matcher(source);
123 		int lastSuff = -1;
124 		while (lexMatcher.find()) {
125 			if (lastSuff < 0)
126 				lastSuff = lexMatcher.start();
127 			parsingBuffer.add( new ParsingBufferElement(
128 					lastSuff,
129 					lexMatcher.end(1) - lexMatcher.start(1),
130 					lexMatcher.end(2) - lexMatcher.start(2),
131 					lexMatcher.group(1),
132 					lexMatcher.start() - lastSuff));
133 			lastSuff = lexMatcher.end(2) - lexMatcher.start(2);
134 		}
135 
136 		Stats.markIt(7);
137 	}

148 	private void resetLAD() {
149 		aliasHash1 = 0;
150 		aliasHash2 = 0;
151 		oldPref=-1;
152 		oldSuff=-1;
153 		this.setNewPrefSufLen(prefix, suffix);
154 	}
155 	/** This method changes the effective lengths of the non-alpha-numeric
156 	 * prefix and suffix. This change affects the results of methods:<br>
157 	 * <code>getAliasHash2</code>, <code>getLength</code>,
158 	 * <code>getAliasStart</code> and <code>getAliasEnd</code>.<br>
159 	 * The last values of the prefix and suffix are stored so if called
160 	 * again with the same values - no recalculations will be performed. 
161 	 * @param prefLen - the new effective prefix length.
162 	 * @param suffLen - the new effective suffix length.
163 	 */
164 	public void setNewPrefSufLen(int prefLen, int suffLen) {
165 		if (oldPref!=prefLen) {
166 			prefLen = Math.min(prefLen, prefix);
167 			int offset = (parsingBuffer.size()>0)?
168 					parsingBuffer.get(frameIx).offset:
169 						0;
170 					aliasOffset1 = offset + prefix - prefLen;
171 		}
172 		if (oldPref!=prefLen || oldSuff!=suffLen) {
173 			suffLen = Math.min(suffLen, suffix);
174 			aliasOffset2 = aliasOffset1 + prefLen + middle + suffLen;
175 
176 			aliasHash2 = 0;  // Recalculation will be done when and if required
177 		}
178 		oldPref=prefLen;
179 		oldSuff=suffLen;
180 	}

185 	/** Method attempts to extend the frame to cover one more parsing-buffer
186 	 * element. If successful - recalculates the frame state metrics.  
187 	 * @return <b>true</b> if the content of the frame was extended with
188 	 * one more buffer element
189 	 */
190 	public boolean parseOne() {
191 		if (!frameCanMove())  return false;
192 
193 		Stats.markIt(-1);
194 		ParsingBufferElement pbe = EMPTY_ELEMENT;
195 		boolean oneParsed = false;
196 		if (restartIx >= 0) {
197 			frameIx = restartIx;
198 			lexemeCount = 0;
199 			oneParsed = (restartIx < parsingBuffer.size());
200 			if (oneParsed)
201 				parsingIx = restartIx;
202 		}
203 		else {
204 			oneParsed = ((parsingIx+1) < parsingBuffer.size());
205 			if (oneParsed)
206 				parsingIx++;
207 		}
208 
209 		if (oneParsed) {
210 			pbe = parsingBuffer.get(parsingIx);
211 			lexemeCount++;
212 		}
213 		Stats.markIt(8);
214 
215 		Stats.markIt(-1);
216 		if (restartIx >= 0) {
217 			prefix = pbe.pref;
218 			middle = pbe.midd;
219 			suffix = pbe.suff;
220 			normalizedAlias = H1_DELIMITER + pbe.middTxt + H1_DELIMITER;
221 		}
222 		else {
223 			middle += suffix + pbe.midd;
224 			suffix = pbe.suff;
225 			normalizedAlias += pbe.middTxt + H1_DELIMITER;
226 		}
227 
228 		// Allows for preliminary detection of parse ending
229 		frameCanExpand = oneParsed && ((parsingIx+1) < parsingBuffer.size());
230 		restartIx = -1;
231 
232 		resetLAD();
233 		Stats.markIt(9);
234 
235 		return oneParsed;
236 	}

238 	/** A shortcut method that extends the frame to cover the whole input
239 	 * string */
240 	public void parseAll() {
241 		while (frameCanExpand)  parseOne();
242 	}

244 	/** The method moves the start of the frame - one buffer element forward
245 	 * and shrinks the size of the frame to one element. */
246 	public void moveOne() {
247 		if (restartIx >= 0)  return;
248 		// Calculate the offset for new parsing start
249 		restartIx = frameIx + 1;
250 
251 		// Conditions under which the frame-movement attempt fails
252 		if (restartIx >= parsingBuffer.size())
253 			restartIx = -1;
254 	}

271 	public int getAliasHash1() {
272 		checkValid();
273 		if (aliasHash1==0)
274 			aliasHash1 = frameTT.transform(normalizedAlias).hashCode(); 
275 		return aliasHash1;

277 	/** Retrieves the <code>HashedAlias</code> related Hash-Code-2. It is

282 	 */

284 		checkValid();
285 		if (aliasHash2==0)
