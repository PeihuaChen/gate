5   /**
6    * This class represents an Entity Alias in the form of couple hash codes.<br>
7    * <br>
8    * For this representation the text is considered a sequence of plain numeric
9    * and letter lexemes delimited by optional non-alpha-numeric character
10   * sequences of arbitrary length. The text of the Alias is considered in two
11   * forms:<br>
12   * "Normalized Form" - the letter and number lexemes of the Alias delimited
13   * by special symbol and concatenated in a string;<br>
14   * "Plain Form" - the string of the Alias without modifications;<br>
15   * The Normalized Form string is used to calculate <code>aliasHash1</code> hash
16   * value and the Plain Form string is used to calculate <code>aliasHash2</code>
17   * hash value. This couple of hash values are used to identify the presence of
18   * the Alias in the parsed text.<br>
19   * The <code>aliasHash1</code> is used for preliminary detection of a possible
20   * Alias location. Because several Aliases can share same value of
21   * <code>aliasHash1</code> - it is not stored as member of the
22   * <code>HashedAlias</code> class, but in the containing structure, which is the
23   * specially designed class <code>HashRegister</code>.<br>
24   * <br>
25   * The <code>aliasHash2</code> hash value is used for final verification and
26   * is stored inside the <code>HashedAlias</code> class. The searched text
27   * is processed lexeme-by-lexeme through the class <code>ParsingFrame</code>.
28   * The <code>aliasHash2</code> is calculated from the plain text of the checked
29   * text fragment. For fast localization of the text fragment which is
30   * candidate for matching the Alias, the fields <code>prefLen</code> and
31   * <code>suffLen</code> are added. They contain the lengths of the
32   * non-alpha-numeric prefix and suffix of the stored Alias. This couple of
33   * values is intended for use with the method
34   * <code>ParsingFrame.setNewPrefSufLen</code><br>
35   * Last two members of the class <code>HashedAlias</code> are the instance
36   * and the semantic class identifiers of the Entity corresponding to the Alias.
37   * They are stored in packed form to save memory.
38   * 
39   * @author danko
40   *
41   */

62      /**
63       * This constructor initializes all final fields of the class and checks
64       * for valid value ranges.
65       * @param aliasHash2 - the hash-code value derived from the plain Alias text
66       * @param prefLen - the size of the non-alpha-numeric prefix of the Alias
67       * @param suffLen - the size of the non-alpha-numeric suffix of the Alias
68       * @param shortInstURI - the compressed instance identifier
69       * @param classID - the encoded semantic class identifier
70       */
71      public HashedAlias(int aliasHash2, int prefLen, int suffLen,
72              String shortInstURI, short classID) {
73          if (prefLen > 127 || suffLen > 127 )
74              throw new RuntimeException("Cannot create HashedAlias with" +
75              		"symbolic prefix/suffix longer than 127!");
76          this.aliasHash2 = aliasHash2;
77          this.prefLen = (byte) prefLen;
78          this.suffLen = (byte) suffLen;
79          this.classID = classID;
80          this.shortInstURI = shortInstURI;
81      }

83      public int compareTo(HashedAlias o) {
84          if (this.prefLen < o.prefLen)  return -1;
85          else if (this.prefLen == o.prefLen && this.suffLen < o.suffLen) return -1;
86          else if (this.prefLen == o.prefLen && this.suffLen == o.suffLen
87                  && this.aliasHash2 < o.aliasHash2) return -1;
88          else if (this.prefLen == o.prefLen && this.suffLen == o.suffLen
89                  && this.aliasHash2 == o.aliasHash2) {
90              // Here null is considered the smallest possible string value 
91              if (this.shortInstURI == null) {
92                  if (o.shortInstURI == null)
93                      return 0;
94                  else
95                      return -1;
96              }
97              else {
98                  if (o.shortInstURI == null)
99                      return 1;
100                 else
101                     return this.shortInstURI.compareTo(o.shortInstURI);
102             }
103         }
104         return 1;
105     }
