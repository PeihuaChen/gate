7   /** This interface allows for an external implementation to feed Entity data
8    * into the KIMGazetteer's Dictionary.
9    * @author danko@sirma.bg
10   *
11   */
12  public interface DictionaryFeeder {
13      //=====================================================================
14      // Method allowing custom initialization of the implementation  
15      //=====================================================================
16      /** This method allows for the KIMGazetteer to pass custom specific
17       * configuration data to the feeder implementation. This data will be
18       * provided as additional configuration parameter for the KIMGazetteer
19       * plug-in. The the gazetteer will pass it to the DictionaryFeeder
20       * implementation at initialization time.
21       * 
22       * @param keyValuePairs - a map of key and value string couples. This
23       * set of parameters will be interpreted only by the DictionaryFeeder
24       * implementation.
25       */
26      void init(Map<String, String> keyValuePairs);
27      
28      //=====================================================================
29      // Method serving the feeding of the complete fixed content Dictionary 
30      //=====================================================================
31      /** This method will be used by the KIMGazetteer to fill its main fixed
32       * dictionary at start-up. If just blank implementation is provided
33       * then no filling will be performed.
34       * 
35       * @param entListener - entry point to pass the data for the main fixed
36       * dictionary
37       */
38      void feedAll(EntityListener entListener);
39  
40      //=====================================================================
41      // Methods serving the feeding of the local temporary Dictionaries  
42      //=====================================================================
43      /** This method marks the start of the current document processing by
44       * the KIMGazetteer. It allows the implementation to perform any document
45       * specific preparations and also feed caching activities.
46       *  
47       * @param document - reference to the document which the gazetteer starts
48       * processing.
49       * @return <b>true</b> - if supports local dictionary data;
50       * <b>false</b> - usage of local dictionaries is disabled;
51       */
52      boolean localFeedInit(Document document);
53  
54      /** This method will be called by the KIMGazetteer's parser. It will be
55       * called after the current parsing step is done, but before a look-up
56       * is performed in the local dictionary.
57       * 
58       * @param document - reference to the document being processed. This is
59       * passed as a reference to the local feeding session in the feeder
60       * implementation.
61       * @param anlList - this list represent the current state of the parsing
62       * mechanism. It should be used to reduce the amount of data that the
63       * feeder will deliver 
64       * @param entListener - entry point to pass the data for the local
65       * dictionary
66       */
67      void localFeedNeeded(Document document, ANLList anlList, EntityListener entListener);
68  
69      /** This method marks the end of the current document processing by
70       * the KIMGazetteer. It allows the implementation to perform any clean-up
71       * activities.
72       * 
73       * @param document - reference to the processed document
74       */
75      void localFeedEnd(Document document);

91          void addEntity(String instUri, String classUri, String aliasLabel);

112         public String getLexeme(int i);
