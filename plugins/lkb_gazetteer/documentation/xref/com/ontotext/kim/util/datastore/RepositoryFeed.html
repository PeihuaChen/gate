23  class RepositoryFeed implements QueryResultListener.Feed {
24  
25  	private final RepositoryConnection conn;
26  	private final Constructor<TupleQueryResultHandler> wrapperFactory;
27  	private final String query;
28  
29  	public RepositoryFeed(RepositoryConnection conn, Constructor<TupleQueryResultHandler> wrapperFactory, String queryString) {			
30  		this.conn = conn;
31  		this.wrapperFactory = wrapperFactory;
32  		this.query = queryString;
33  	}
34  
35  	public void feedTo(QueryResultListener listener)
36  	throws KIMQueryException {
37  		TupleQueryResultHandler handler = new ListenerAdapter(listener);
38  		if (wrapperFactory != null) {
39  			try {
40  				handler = wrapperFactory.newInstance(handler);
41  			} catch (Exception e) {
42  				e.printStackTrace();
43  			}
44  		}
45  
46  		try {			 
47  			
48  			// Note that conn.prepareTupleQuery(ql, query) may not actually parse the query 
49  			// if the repository is remote.
50  			
51  			QueryLanguage lang = findLanguage(query);
52  			TupleQuery tq = conn.prepareTupleQuery(lang, query);
53  			tq.evaluate(handler);
54  		} catch (Exception e) {
55  			throw new KIMQueryException(e, query);
56  		}
57  	}
58  
59  	private final QueryLanguage findLanguage(String query) throws MalformedQueryException {
60  		String message = "";
61  		for (QueryParserFactory qp : Arrays.asList(new SPARQLParserFactory(), new SeRQLParserFactory())) {
62  			try {
63  				qp.getParser().parseQuery(query, WKBConstants.WKB_NS);
64  				return qp.getQueryLanguage();
65  			} catch (MalformedQueryException e) {
66  				message += e.getMessage() + "\n";
67  			}
68  		}
69  		throw new MalformedQueryException(message);
70  	}
