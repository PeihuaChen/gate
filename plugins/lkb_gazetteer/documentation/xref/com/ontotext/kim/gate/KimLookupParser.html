9   /**
10   * This class processes a textual input, finds the location of all known text
11   * fragments stored in a dictionary and then passes these findings to a special
12   * handler.<br>
13   * <br>
14   * The text processing is performed through the class <code>ParsingFrame</code>
15   * .<br>
16   * An instance of class <code>AliasCacheImpl</code> is used for dictionary.<br>
17   * The findings are passed to a handler class that implements the interface
18   * <code>EntityOccuranceHandler</code>.<br>
19   * 
20   * @author danko
21   *
22   */
23  public class KimLookupParser {
24  
25      private static Logger log = Logger.getLogger(KimLookupParser.class);
26      
27      public interface EntityOccuranceHandler {
28          void processEntityOccurance(int start, int end,
29                  String instURI, String classURI);
30      }
31      
32      public interface AliasLookupDictionary {
33          /**
34           * Looks up for matches given a ParsingFrame. This is used for
35           * multiple lookups for different fragments of a parsed text.
36           * @param pfm - a ParsingFrame which has already parsed a part
37           * of the text
38           * @return - collection of matching wrapped Aliases
39           */
40          public Collection<KimLookupParser.AliasWrapper> lookup(ParsingFrame pfm);
41          /**
42           * Checks if the lexeme phrase that is currently focused by the
43           * passed ParsingFrame is a valid lexeme prefix of another alias.
44           * If it is a valid prefix - then phrase can grow.
45           * @param pfm - a ParsingFrame which has already parsed a part
46           * of the text
47           * @return - true is there is still a chance to find a larger 
48           * phrase matching the current location of the parsed text
49           */
50          public boolean canPhraseGrow(ParsingFrame pfm);
51      }
52      
53      /** This class implements a container to return the results of the
54       * Alias Dictionary lookup */
55      public static class AliasWrapper {
56      	public final String instURI;  // The instance URI of the related Entity
57      	public final String classURI;  // The class URI of the related Entity
58      	public final int start;
59      	public final int end;
60      	public AliasWrapper(String instURI, String classURI,
61      			int start, int end) {
62      		this.instURI = instURI; this.classURI = classURI;
63      		this.start = start; this.end = end;
64      	}
65      }

67      private AliasLookupDictionary aliasDictionary;
68      private boolean interrupted = false;
69  
70      public KimLookupParser(AliasLookupDictionary aliasCache) {
71          this.aliasDictionary = aliasCache;
72      }
73      
74      public void findLookups(String content, EntityOccuranceHandler entityHandler) {
75          ParsingFrame pfm = new ParsingFrame(content);
76          
77          log.debug("Time tracing begins");     
78          
79          Collection<KimLookupParser.AliasWrapper> currentMatch;
80          do {
81              if (pfm.parseOne()) {
82                  currentMatch = aliasDictionary.lookup(pfm);
83                  if (currentMatch != null) {
84                      for (KimLookupParser.AliasWrapper ent : currentMatch) {
85                          entityHandler.processEntityOccurance(ent.start, ent.end, ent.instURI, ent.classURI);
86                      }
87                  }
88              }
89              if (!aliasDictionary.canPhraseGrow(pfm) || !pfm.frameCanExpand())
90                  pfm.moveOne();
91          } while (pfm.frameCanMove());
92          log.debug("Time tracing ends");       
93      }
